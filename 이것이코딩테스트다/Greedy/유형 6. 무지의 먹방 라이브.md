```java
class Solution {
    public int solution(int[] food_times, long k) {
        int answer = 0;
        int flagCnt = 0;
        int total = 0;
        
        for (int i = 0; i < food_times.length; i++) {
             total += food_times[i];
        }
        if(total <= k ) return -1;
        
        while(true) {
            for (int i = 0; i < food_times.length; i++) {
                flagCnt++;

                if (k == flagCnt) {
                    answer = i;
                    return answer;
                }
                if (food_times[i] == 0) {
                    
                    continue;
                }
                food_times[i]--;
            }
        }
    }
}
```
